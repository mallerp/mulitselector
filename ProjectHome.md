JQuery Pop-up checkbox, this plug in use jquery 1.3.2 library.

example code ：
```

	var data = 
	[
		{id: "10",name: "请选择中国苹果"},
		{id: "20",name: "请选择中国香蕉"},
		{id: "30",name: "请选择中国西瓜"},
		{id: "40",name: "请选择中国桃子"},
		{id: "50",name: "请选择中国葡萄"}
	];

	$("#test1").mulitselector({
		title:"请选择类别-自定义标题",
		width:"530px",
		data:data
	});
```
